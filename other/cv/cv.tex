\subsection{Research}
\cventry{2017--now}{Postdoctoral Research}{\Skoltech}{\Moscow}{}{
\begin{itemize}
    \item \textbf{Multiscale multiphysics modeling and simulation of laser powder bed fusion}
        (funded by \underline{Oerlikon AM} in 2018--2022):
    \begin{itemize}
        \item Developed a set of computational tools for large-scale detailed simulations of a wide range
        of physical processes occurring in the laser powder-bed fusion technology
        (within the OpenFOAM environment).
        \item Developed a thermo-fluid-dynamic model of powder particle fusion for high-fidelity
        simulations of the melt-pool dynamics including a ray-tracing-based algorithm for evaluating
        absorptivity of the laser energy.
        \item Developed a multi-component phase-field model of dendritic alloy solidification
        for predicting subgrain structure and microsegregation patterns in 3D-printed materials
        from commercial alloys.
        \item Developed a thermo-mechanical model for predicting residual macroscopic stresses
        in parts produced by layer-by-layer 3D printing.
    \end{itemize}
    \item \textbf{Macroscopic modeling of ceramic additive manufacturing}
        (funded by \underline{Ministry of Science and Higher Education grant No.~05.618.21.0006} in 2020):
    \begin{itemize}
        \item Developed a mechanical model of a green body for predicting residual stresses occurring
        in ceramic stereolithography technologies.
        \item Developed a fluid-dynamic model of polymer binder removal from green parts produced
        by ceramic stereolithography for predicting critical conditions of thermal debinding.
    \end{itemize}
    \item \textbf{Large-scale simulations of gaseous detonation in complex geometries}
        (funded by \underline{RFBR grant No.~20-51-00004} in 2020--2022
        and \href{https://rscf.ru/project/22-21-00902/}{\underline{RScF grant No.~22-21-00902}} in 2022--2023):
    \begin{itemize}
        \item Developed an OpenFOAM solver for reactive compressible viscous flows for unsteady
        three-dimensional simulations of gaseous detonation based on the one-step kinetics.
        \item Studied numerically the influence of momentum and energy losses on the detonation
        wave dynamics in a rectangular channel with obstacles.
    \end{itemize}
\end{itemize}
}
\cventry{2017--now}{Postdoctoral Research}{\Fic}{\Moscow}{}{
\begin{itemize}
    \item \textbf{High-accuracy simulations of rarefied gas flows on the basis of the Boltzmann equation}
    (funded by \underline{RFBR grant No.~18-07-01500-a} in 2018--2020):
    \begin{itemize}
        \item Validated the original solver for the Boltzmann equation against the computational results,
        obtained by the DSMC method and BGK-type model equations~\cite{rogozin2019regularization}.
    \end{itemize}
    \item \textbf{Kinetic-based numerical methods for multiscale gas flows}
    (funded by \underline{RFBR grant No.~18-01-00899-a} in 2018--2020):
    \begin{itemize}
        \item Developed a novel hybrid computational method based on the discrete-velocity and
        lattice-Boltzmann methods~\cite{aristov2020kinetic, aristov2019hybrid}.
    \end{itemize}
\end{itemize}
}
